**ORDERS_CAT**

INSERT INTO orders_cat(Column_name, Value_in_column, Count_value)
SELECT 'orderDate',  orderDate, COUNT(*) 
	FROM orders 
	GROUP BY orderDate;
    
INSERT INTO orders_cat(Column_name, Value_in_column, Count_value)
SELECT 'requiredDate',  requiredDate, COUNT(*) 
	FROM orders 
	GROUP BY requiredDate;
    
INSERT INTO orders_cat(Column_name, Value_in_column, Count_value)

	FROM orders 
	GROUP BY shippedDate;
