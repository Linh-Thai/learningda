**CUSTOMERS_NUMERIC**

INSERT INTO customers_numeric(Column_name, Min_value, Max_value, Avg_value, STDDEV_value)
SELECT 'creditLimit', MIN(creditLimit), MAX(creditLimit), AVG(creditLimit), STDDEV(creditLimit) 
FROM customers;
